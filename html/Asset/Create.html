<%args> $id => "" <%init> my $asset = RT::Asset->new( $session{CurrentUser} ); Abort("You don't have permission to create assets.") unless $asset->CurrentUserHasRight("CreateAsset"); my @results; if ($id eq "new") { my $skip_create = 0; my ($cf_ok, @cf_errors) = $m->comp( '/Elements/ValidateCustomFields', CustomFields => $asset->CustomFields, NamePrefix => "Object-RT::Asset--CustomField-", ARGSRef => \%ARGS ); $m->callback( %ARGS, CallbackName => 'BeforeCreate', AssetObj => $asset, results => \@results, cf_ok => \$cf_ok, skip_create => \$skip_create, ); if ($cf_ok and not $skip_create) { my %create = ProcessObjectCustomFieldUpdatesForCreate( ARGSRef => \%ARGS ); $create{$_} = $ARGS{$_} for qw(Name Description); $m->callback( %ARGS, AssetObj => $asset, Create => \%create, CallbackName => 'MassageCreate' ); my ($ok, $msg) = $asset->Create( %create ); if ($ok) { MaybeRedirectForResults( Actions => [$msg], Path => "/Asset/Display.html", Arguments => { id => $asset->id }, ); } else { # Report the error but don't redirect away so they can re-submit push @results, $msg; } } else { push @results, @cf_errors; } }
